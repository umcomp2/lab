def hijo_rojo(pipe):

    lista_rojo = []
    archivo = open("Rojo.txt", "wt")
    while True:
        a = pipe.recv()
        if a == b'EOF':
            break
        for pixel in a:
            if len(a) % 3 == 0:
                lista_rojo.append(pixel)
    
    archivo.write(str(dict(enumerate(lista_rojo))))

    pipe.close()
    archivo.close()
